

设置环境变量：

#JAVA_HOME
export JAVA_HOME=/hadoop/jdk1.8.0_144
export PATH=$JAVA_HOME/bin:$PATH

#HADOOP_HOME
export HADOOP_HOME=/hadoop/hadoop-2.7.7
export HADOOP_CONF_DIR=/hadoop/hadoop-2.7.7/etc/hadoop
export PATH=$PATH:$HADOOP_HOME/bin
export PATH=$PATH:$HADOOP_HOME/sbin
export PATH=$PATH:$HADOOP_CONF_DIR


创建：
/Applications/hadoop-2.7.7/tmp

core-site.xml
添加：
<configuration>
<property>
<name>fs.defaultFS</name>
<value>hdfs://localhost:9000</value>
</property>
<property>
<name>hadoop.tmp.dir</name>
<value>/Applications/hadoop-2.7.7/tmp</value>
</property>
</configuration>



hdfs-site.xml

添加：
<configuration>
<property>
<name>dfs.replication</name>
<value>1</value>
</property>
</configuration>



yarn-site.xml
添加：
<configuration>
<property>
<name>yarn.resourcemanager.address</name>
<value>127.0.0.1:8032</value>
</property>
<property>
<name>yarn.resourcemanager.scheduler.address</name>
<value>127.0.0.1:8030</value>
</property>
<property>
<name>yarn.resourcemanager.resource-tracker.address</name>
<value>127.0.0.1:8031</value>
</property>
<property>
<name>yarn.nodemanager.aux-services</name>
<value>mapreduce_shuffle</value>
</property>
<property>
<name>yarn.resourcemanager.hostname</name>
<value>localhost</value>
</property>
</configuration>

mapred-site.xml
添加：
<configuration>
<property>
<name>mapreduce.framework.name</name>
<value>yarn</value>
</property>
</configuration>



hadoop-env.sh
yarn-env.sh
mapred-env.sh
添加：
export JAVA_HOME=/hadoop/jdk1.8.0_144




格式化：
hdfs namenode -format

启动：
sbin/start-all.sh

sbin/stop-all.sh
